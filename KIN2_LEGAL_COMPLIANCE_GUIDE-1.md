     async handleAccessRequest(userId) {

       const data = await this.exportUserData(userId);

### Background Check Compliance (FCRA - US)

2. **Adverse Action Process:**

3. **Implementation:**
   ```javascript
   // backend/services/compliance/BackgroundCheckService.js
   
   class BackgroundCheckService {
     async requestCheck(workerId, employerId) {
       // 1. Verify worker consent obtained
       const consent = await this.verifyConsent(workerId);
       if (!consent) {
         throw new Error('FCRA consent required');
       }
       
       // 2. Provide disclosure
       await this.sendFCRADisclosure(workerId);
       
       // 3. Request from provider
       const report = await this.backgroundCheckProvider.request({
         subjectId: workerId,
         requestorId: employerId
       });
       
       return report;
     }
     
     async adverseAction(workerId, employerId, report) {
       // Pre-adverse action
       await this.sendPreAdverseActionNotice(workerId, {
         report,
         fcraRights: true
       });
       
       // Wait period (5 business days)
       await this.waitPeriod(5);
       
       // Final adverse action
       await this.sendFinalAdverseActionNotice(workerId);
     }
   }
   ```

    }

    }
  }

    }
  }
